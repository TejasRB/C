#include <stdlib.h>
#include <conio.h>
#include <stdio.h>

void main() {
    int a[100], a1[10] = {0}, a2[100], i, result, n;  // Initialize a1 array with zeros
    
    printf("Enter the number of elements: ");
    scanf("%d", &n);
    
    printf("Enter the elements\n");
    for (i = 0; i < n; i++)
        scanf("%d", &a[i]);
    
    for (i = 0; i < n; i++) {
        result = a[i] % 10;
        a1[result] = a[i];  // Store the value in the correct index of a1 array
        a2[i] = a[i];
        printf("Location: a1[%d], value: %d\n", result, a1[result]);
    }
    
    printf("Hash table:\n");
    for (i = 0; i < n; i++)
        printf("%d ", a2[i]);
    
    getch();
}
